% Class %
% KOMA-Script's scrbook (see https://ctan.org/pkg/koma-script)
% Book/paper size: 170 mm x 240 mm; approx. 0.81 * A4 (see guidelines)

paper=170mm:240mm,pagesize=auto, % format and setting correct page values
fontsize=10pt,twoside, % Font size and sided-ness

headinclude=true,footinclude=false,mpinclude=false, % consideration of header/footer/marginalia in text body
headsepline=true,footsepline=false, % separation line between header/footer and text
]{scrbook}


% Fundamentals %
% Settings
%% Remove the restriction of aligning the bottom-most material between pages.
%% While enforcing this restriction (by means of \flushbottom) is typically
%% viewed as the typographically correct approach for books, it creates a lot of
%% unclean vertical stretching, especially when there is a lot of
%% displayed/floating material. Synchronizing and coordinating the material in a
%% way that removes (most) of this stretching, although probably possible, may
%% be unrealistically arduous relative to the gain. As such, we opt to remove
%% this restriction here.
\raggedbottom

%%% Number and title on a single line (chapterprefix=false)
\makeatletter
\renewcommand\chapterlinesformat[3]{%
  \Ifstr{#1}{chapter}{% Only format differently for chapter level (as you can reuse chapter style for other levels)
    % Align chapter title default, number lowered in margin
    \raggedchapter#3\marginline{\usekomafont{chapter}\raisebox{-\height-1.25\baselineskip}{#2}}%
    \par\nobreak\vspace{0.2\baselineskip-\parskip}\titlerule% Draw title rule
  }{%
    \@hangfrom{#2}{#3}% Default definition
  }
}
\makeatother

%%% Number and title on different lines (chapterprefix=true)
\renewcommand{\chapterlineswithprefixformat}[3]{%
  \Ifstr{#1}{chapter}{% Only format differently for chapter level (as you can reuse chapter style for other levels)
    \raggedleft#2%
    \raggedchapter#3% Align title the default way and align number right
    \par\nobreak\vspace{0.2\baselineskip-\parskip}\titlerule%
  }{%
    #2#3% Default definition
  }
}

  \normalfont\normalcolor\footnotesize%

}

font=\marginparnotefont,
]{marginpar}

%%% Margin note without footnote marker or label
\NewDocumentCommand{\marnote}{s O{marginpar} m}{%
  \IfBooleanTF{#1}{% Regular (expanded)
    \makenote[#2]{%
      #3%
    }%
  }{% Starred (detokenized)
    \makenote*[#2]{%
      #3%
    }%
  }%
}

%%% Margin note with footnote marker and (optional) label, updates footnote counter
\NewDocumentCommand{\marfootnote}{O{marginpar} m o}{%
  \IfValueTF{#3}{%
    \refstepcounter{footnote}%
    \label{#3}%
  }{%
    \stepcounter{footnote}%
  }%
  \textsuperscript{\thefootnote}%
  \makenote[#1]{%
    \textsuperscript{\thefootnote}~#2%
  }%
}%
